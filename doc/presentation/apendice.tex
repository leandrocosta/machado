\section{Exemplo de Execução da Heurística de Obtenção de Conjuntos Ortogonais}

\begin{frame}
	\begin{itemize}[<+-| alert@+>]
		\item $\F = \{A, B, BC, ABC\}$
		\item $Sort (\F)$
		\item $\F = \{ABC, BC, A, B\}$
	\end{itemize}
\end{frame}
