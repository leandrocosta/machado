\section{Conclusão}
\subsection{Resultados}

\begin{frame}{Acurácia}
	\begin{itemize}[<+-| alert@+>]
		\item As abordagens baseadas em ortogonalidade obtiveram resultados semelhantes aos da abordagem clássica:
		\begin{itemize}[<+-| alert@+>]
			\item Considerando os melhores parâmetros para cada base, as médias de acurácia obtidas para as abordagens LAC, OLAC e ORIGAMI foram, respectivamente, $0.843$, $0.840$ e $ 0.839$;
			\item Considerando os melhores parâmetros para a média dos resultados, as médias de acurácia obtidas para as abordagens LAC, OLAC e ORIGAMI foram, respectivamente, $0.808$, $0.813$ e $0.782$.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Padrões}
	\begin{itemize}[<+-| alert@+>]
		\item A quantidade de padrões utilizados na geração das regras nas abordagens ortogonais foi bem menor que na abordagem clássica:
		\begin{itemize}[<+-| alert@+>]
			\item Considerando os melhores parâmetros para cada base, as quantidades médias de padrões utilizados pelas abordagens LAC, OLAC e ORIGAMI foram, respectivamente, $213$, $12$ e $12$;
			\item Considerando os melhores parâmetros para a média dos resultados, as quantidades de padrões utilizados pelas abordagens LAC, OLAC e ORIGAMI foram, respectivamente, $19$, $12$ e $1$.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Regras}
	\begin{itemize}[<+-| alert@+>]
		\item Conseqüentemente, a quantidade de regras geradas nas abordagens ortogonais também foi menor que na abordagem clássica:
		\begin{itemize}[<+-| alert@+>]
			\item Considerando os melhores parâmetros para cada base, as quantidades médias de regras geradas pelas abordagens LAC, OLAC e ORIGAMI foram, respectivamente, $628$, $25$ e $23$;
			\item Considerando os melhores parâmetros para a média dos resultados, as quantidades de regras geradas pelas abordagens LAC, OLAC e ORIGAMI foram, respectivamente, $51$, $31$ e $1$.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Outros Resultados}
	\begin{itemize}[<+-| alert@+>]
		\item A métrica de ortogonalidade baseada na estrutura dos padrões obteve melhores resultados;
		\item As métricas para avaliação de Regras Associativas que obtiveram melhores resultados foram convicção para LAC e OLAC e confiança para ORIGAMI;
		\item A maior parte das falhas na classificação baseada em ortogonalidade foi causada pela característica dos padrões, e não pela baixa medida de ortogonalidade no conjunto.
	\end{itemize}
\end{frame}
