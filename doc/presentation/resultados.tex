\subsection{Experimentos}

\begin{frame}{O Aplicativo \textbf{olac}}
	\begin{itemize}[<+-| alert@+>]
		\item O aplicativo \textbf{olac} possui a implementação de três abordagens distintas de um classificador baseado em regras de associação:
		\begin{itemize}[<+-| alert@+>]
			\item A abordagem LAC (\textit{Lazy Associative Classifier}), é a abordagem \textit{lazy} na sua versão original (e não-ortogonal);
			\item A abordagem OLAC (\textit{Orthogonal Lazy Associative Classifier}) é a modificação da abordagem \textit{lazy} que considera a ortogonalidade dos padrões durante a tarefa de obtenção de regras;
			\item A abordagem ORIGAMI é a implementação da adaptação apresentada para a estratégia ORIGAMI.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Metodologia}
	\begin{itemize}[<+-| alert@+>]
		\item Foram utilizadas 26 bases de dados do repositório \textbf{UCI} (\textit{UC Irvine Machine Learning Repository}), amplamente referenciado em pesquisas na área de classificação em mineração de dados;
		\item Todas as bases utilizadas durante os testes foram reordenadas aleatoriamente e particionadas em dez sub-conjuntos, de onde foram criadas dez configurações de teste para cada uma delas;
		\item Cada configuração de teste consiste de uma parte (um sub-conjunto da base) como arquivo de teste, e nove partes (os nove sub-conjuntos restantes da base) como arquivo de treinamento;
			\end{itemize}
\end{frame}

\begin{frame}{Metodologia}
	\begin{itemize}[<+-| alert@+>]
		\item Como resultados foram considerados a média das dez execuções diferentes para cada base de dados;
		\item Os parâmetros utilizados nos testes se encontram na tabela \ref{tab:table_test_parms};
		\item Todas as combinações possíveis destes parâmetros foram realizadas, com exceção da combinação tamanho máximo de regra  $1$ e métrica de ortogonalidade $s$ para o OLAC.
	\end{itemize}
\end{frame}
